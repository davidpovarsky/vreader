## Commands run

```bash
# Build and install
DEVELOPER_DIR=/Applications/Xcode.app/Contents/Developer xcodebuild build \
  -project vreader.xcodeproj -scheme vreader \
  -destination 'platform=iOS Simulator,name=iPhone 17 Pro' \
  -configuration Debug
APP_PATH=$(find ~/Library/Developer/Xcode/DerivedData/vreader-hdhlhcqmxppsadhececcxeadpkvz/Build/Products/Debug-iphonesimulator -name "vreader.app" -maxdepth 2 | head -1)
xcrun simctl install booted "$APP_PATH"

# Grant debug scheme approval
bash scripts/grant-debug-scheme-approval.sh

# Reset and seed
xcrun simctl openurl booted 'vreader-debug://reset'
xcrun simctl openurl booted 'vreader-debug://seed?fixture=war-and-peace'

# Open book
SIM_ID="1FAB9493-B97E-48F0-96C7-44A8E5AAA21E"
DATA=$(xcrun simctl get_app_container "$SIM_ID" com.vreader.app data)
FILE=$(ls -t "$DATA/Library/Application Support/ImportedBooks/" | head -1)
KEY=$(echo "$FILE" | sed -E 's/^([a-z0-9]+)_([0-9a-f]+)_([0-9]+)\..*/\1:\2:\3/')
ENCODED=$(printf '%s' "$KEY" | sed 's/:/%3A/g')
xcrun simctl openurl "$SIM_ID" "vreader-debug://open?bookId=$ENCODED"

# Settle and snapshot
xcrun simctl openurl "$SIM_ID" 'vreader-debug://settle?token=wi2-verify'
xcrun simctl openurl "$SIM_ID" 'vreader-debug://snapshot?dest=wi2-state.json'

# Log stream (filtered) during open to capture chapter count
xcrun simctl spawn "$SIM_ID" log stream --predicate 'subsystem == "com.vreader.app"' --debug
```
